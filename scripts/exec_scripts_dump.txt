	centerlatlog = cg.drag(dx,dy,[centerLat,centerLon],misdatos.par.value7)
	#print(centerlatlog)

	misdatos.par.value11 = centerlatlog[1]

	norm = cg.fromLngLatToTileIndex([centerLat, centerLon])

	centerTileIndexY = math.floor(norm[1] * fixedPowZoom)
	misdatos.par.value4=centerTileIndexX
	misdatos.par.value5=centerTileIndexY
	return
